// api/sbg-rules.js (Vercel Serverless Function) // ----------------------------------------------------------- // Stores + serves rules/lenders JSON via Vercel Blob storage. // Env needed: // BLOB_READ_WRITE_TOKEN (from Vercel > Storage > Blob) // RULES_ADMIN_KEY (string you share with admins) // ----------------------------------------------------------- import { put, list, del, head, get } from '@vercel/blob'; const KEY = process.env.RULES_ADMIN_KEY; const BUCKET_FILE = 'sbg-rules.json'; // single blob name export default async function handler(req, res){ if (req.method === 'GET'){ try{ const file = await get(`public/${BUCKET_FILE}`); if(!file) return res.status(200).json({ rules: [], lenders: [] }); const text = await file.text(); const json = JSON.parse(text); return res.status(200).json({ rules: json.rules||[], lenders: json.lenders||[] }); }catch(e){ return res.status(200).json({ rules: [], lenders: [] }); } } if (req.method === 'PUT'){ try{ if(!KEY || req.headers['x-admin-key'] !== KEY) return res.status(401).json({ error: 'Unauthorized' }); const body = req.body || {}; const rules = Array.isArray(body.rules)? body.rules : []; const lenders = Array.isArray(body.lenders)? body.lenders : []; // lightweight shape checks if(!Array.isArray(rules) || !Array.isArray(lenders)) return res.status(400).json({ error:'Invalid payload' }); const payload = JSON.stringify({ rules, lenders }, null, 2); await put(`public/${BUCKET_FILE}`, payload, { access: 'public' }); return res.status(200).json({ ok:true }); }catch(e){ return res.status(500).send(String(e?.message||e)); } } res.status(405).send('Method not allowed'); }
