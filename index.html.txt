        async function analyzeMessage() {
            const messageText = document.getElementById('message').value.trim();
            
            if (!messageText) {
                alert('Please enter a message to analyze.');
                return;
            }

            // Show loading state
            document.getElementById('loading').style.display = 'block';
            document.getElementById('results').style.display = 'none';
            document.querySelector('.check-btn').disabled = true;

            // Simulate AI analysis delay
            await new Promise(resolve => setTimeout(resolve, 1500));

            const analysis = performComplianceCheck(messageText);
            displayResults(analysis);

            // Hide loading state
            document.getElementById('loading').style.display = 'none';
            document.querySelector('.check-btn').disabled = false;
        }

        function displayResults(analysis) {
            const resultsDiv = document.getElementById('results');
            const isCompliant = analysis.isCompliant;
            
            resultsDiv.className = `results ${isCompliant ? 'compliant' : 'non-compliant'}`;
            
            const iconSvg = isCompliant ? 
                '<svg class="icon" fill="currentColor" viewBox="0 0 20 20"><path fill-rule="evenodd" d="M10 18a8 8 0 100-16 8 8 0 000 16zm3.707-9.293a1 1 0 00-1.414-1.414L9 10.586 7.707 9.293a1 1 0 00-1.414 1.414l2 2a1 1 0 001.414 0l4-4z" clip-rule="evenodd"></path></svg>' :
                '<svg class="icon" fill="currentColor" viewBox="0 0 20 20"><path fill-rule="evenodd" d="M18 10a8 8 0 11-16 0 8 8 0 0116 0zm-7 4a1 1 0 11-2 0 1 1 0 012 0zm-1-9a1 1 0 00-1 1v4a1 1 0 102 0V6a1 1 0 00-1-1z" clip-rule="evenodd"></path></svg>';

            let html = `
                <h3>
                    ${iconSvg}
                    ${isCompliant ? 'Message is 10DLC Compliant!' : 'Compliance Issues Found'}
                </h3>
                <p><strong>Message Length:</strong> ${analysis.messageLength} characters (${analysis.wordCount} words)</p>
            `;

            // Display detected restricted categories
            if (analysis.detectedCategories && analysis.detectedCategories.length > 0) {
                html += '<div class="category-warning">';
                html += '<h4>‚ö†Ô∏è Detected Issues - Campaign Impact Analysis</h4>';
                
                // Create summary badges
                html += '<div class="category-summary">';
                analysis.detectedCategories.forEach(category => {
                    const impactClass = category.impact.includes('PROHIBITED') ? 'prohibited' : 
                                       category.impact.includes('RESTRICTED') ? 'restricted' : 
                                       category.impact.includes('BRANDING') ? 'branding' : 'warning';
                    
                    html += `<span class="category-badge ${impactClass}">${category.name}</span>`;
                });
                html += '</div>';
                
                html += `<button class="toggle-details" onclick="toggleCategoryDetails(this)">Show Details</button>`;
                
                html += '<div class="category-details">';
                analysis.detectedCategories.forEach(category => {
                    html += `
                        <div class="category-item">
                            <div class="category-title">${category.name}</div>
                            <p><strong>Impact:</strong> ${category.impact.split(' - ')[1] || category.impact}</p>
                            <div class="category-keywords">Keywords found: "${category.keywords.join('", "')}"</div>
                        </div>
                    `;
                });
                html += '</div>';
                html += '</div>';
            }

            if (analysis.issues.length > 0) {
                // Create issue summary
                const highIssues = analysis.issues.filter(issue => issue.severity === 'high').length;
                const mediumIssues = analysis.issues.filter(issue => issue.severity === 'medium').length;
                const lowIssues = analysis.issues.filter(issue => issue.severity === 'low').length;

                html += '<div class="issue-header">';
                html += '<div class="issue-summary">';
                if (highIssues > 0) html += `<span class="issue-count high">${highIssues} Critical</span>`;
                if (mediumIssues > 0) html += `<span class="issue-count medium">${mediumIssues} Medium</span>`;
                if (lowIssues > 0) html += `<span class="issue-count low">${lowIssues} Minor</span>`;
                html += '</div>';
                html += '<button class="toggle-issues" onclick="toggleIssueDetails(this)">Show Issues</button>';
                html += '</div>';

                html += '<ul class="issue-list">';
                analysis.issues.forEach(issue => {
                    html += `
                        <li class="${issue.severity}">
                            <div class="issue-message">${issue.message}</div>
                            ${issue.suggestion ? `<div class="issue-suggestion">${issue.suggestion}</div>` : ''}
                        </li>
                    `;
                });
                html += '</ul>';
            } else if (analysis.detectedCategories.length === 0) {
                html += '<p>‚úÖ No compliance issues detected! Your message appears to follow 10DLC guidelines.</p>';
            }

            resultsDiv.innerHTML = html;
            resultsDiv.style.display = 'block';
        }

        function toggleIssueDetails(button) {
            const issueList = button.parentElement.nextElementSibling;
            const isShowing = issueList.classList.contains('show');
            
            if (isShowing) {
                issueList.classList.remove('show');
                button.textContent = 'Show Issues';
            } else {
                issueList.classList.add('show');
                button.textContent = 'Hide Issues';
            }
        }

        function toggleCategoryDetails(button) {
            const details = button.nextElementSibling;
            const isShowing = details.classList.contains('show');
            
            if (isShowing) {
                details.classList.remove('show');
                button.textContent = 'Show Details';
            } else {
                details.classList.add('show');
                button.textContent = 'Hide Details';
            }
        }

        function toggleCollapse(header, listId) {
            const list = document.getElementById(listId);
            const isExpanded = list.classList.contains('expanded');
            
            if (isExpanded) {
                list.classList.remove('expanded');
                header.classList.remove('expanded');
            } else {
                list.classList.add('expanded');
                header.classList.add('expanded');
            }
        }

        // Allow Enter key to trigger analysis (with Shift+Enter for new lines)
        document.getElementById('message').addEventListener('keydown', function(e) {
            if (e.key === 'Enter' && !e.shiftKey) {
                e.preventDefault();
                analyzeMessage();
            }
        });
